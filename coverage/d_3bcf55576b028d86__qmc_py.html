90def scale( 

91 sample: npt.ArrayLike, 

92 l_bounds: npt.ArrayLike, 

93 u_bounds: npt.ArrayLike, 

94 *, 

95 reverse: bool = False 

96) -> np.ndarray: 

97 r"""Sample scaling from unit hypercube to different bounds. 

98 

99 To convert a sample from :math:`[0, 1)` to :math:`[a, b), b>a`, 

100 with :math:`a` the lower bounds and :math:`b` the upper bounds. 

101 The following transformation is used: 

102 

103 .. math:: 

104 

105 (b - a) \cdot \text{sample} + a 

106 

107 Parameters 

108 ---------- 

109 sample : array_like (n, d) 

110 Sample to scale. 

111 l_bounds, u_bounds : array_like (d,) 

112 Lower and upper bounds (resp. :math:`a`, :math:`b`) of transformed 

113 data. If `reverse` is True, range of the original data to transform 

114 to the unit hypercube. 

115 reverse : bool, optional 

116 Reverse the transformation from different bounds to the unit hypercube. 

117 Default is False. 

118 

119 Returns 

120 ------- 

121 sample : array_like (n, d) 

122 Scaled sample. 

123 

124 Examples 

125 -------- 

126 Transform 3 samples in the unit hypercube to bounds: 

127 

128 >>> from scipy.stats import qmc 

129 >>> l_bounds = [-2, 0] 

130 >>> u_bounds = [6, 5] 

131 >>> sample = [[0.5 , 0.75], 

132 ... [0.5 , 0.5], 

133 ... [0.75, 0.25]] 

134 >>> sample_scaled = qmc.scale(sample, l_bounds, u_bounds) 

135 >>> sample_scaled 

136 array([[2. , 3.75], 

137 [2. , 2.5 ], 

138 [4. , 1.25]]) 

139 

140 And convert back to the unit hypercube: 

141 

142 >>> sample_ = qmc.scale(sample_scaled, l_bounds, u_bounds, reverse=True) 

143 >>> sample_ 

144 array([[0.5 , 0.75], 

145 [0.5 , 0.5 ], 

146 [0.75, 0.25]]) 

147 

148 """ 

149 sample = np.asarray(sample) 

150 lower = np.atleast_1d(l_bounds) 

151 upper = np.atleast_1d(u_bounds) 

152 

153 # Checking bounds and sample 

154 if not sample.ndim == 2: 

155 raise ValueError('Sample is not a 2D array') 

156 

157 lower, upper = np.broadcast_arrays(lower, upper) 

158 

159 if not np.all(lower < upper): 

160 raise ValueError('Bounds are not consistent a < b') 

161 

162 if len(lower) != sample.shape[1]: 

163 raise ValueError('Sample dimension is different than bounds dimension') 

164 

165 if not reverse: 

166 # Checking that sample is within the hypercube 

167 if not (np.all(sample >= 0) and np.all(sample <= 1)): 

168 raise ValueError('Sample is not in unit hypercube') 

169 

170 return sample * (upper - lower) + lower 

171 else: 

172 # Checking that sample is within the bounds 

173 if not (np.all(sample >= lower) and np.all(sample <= upper)): 

174 raise ValueError('Sample is out of bounds') 

175 

176 return (sample - lower) / (upper - lower)

179def discrepancy( 

180 sample: npt.ArrayLike, 

181 *, 

182 iterative: bool = False, 

183 method: Literal["CD", "WD", "MD", "L2-star"] = "CD", 

184 workers: IntNumber = 1) -> float: 

185 """Discrepancy of a given sample. 

186 

187 Parameters 

188 ---------- 

189 sample : array_like (n, d) 

190 The sample to compute the discrepancy from. 

191 iterative : bool, optional 

192 Must be False if not using it for updating the discrepancy. 

193 Default is False. Refer to the notes for more details. 

194 method : str, optional 

195 Type of discrepancy, can be ``CD``, ``WD``, ``MD`` or ``L2-star``. 

196 Refer to the notes for more details. Default is ``CD``. 

197 workers : int, optional 

198 Number of workers to use for parallel processing. If -1 is given all 

199 CPU threads are used. Default is 1. 

200 

201 Returns 

202 ------- 

203 discrepancy : float 

204 Discrepancy. 

205 

206 Notes 

207 ----- 

208 The discrepancy is a uniformity criterion used to assess the space filling 

209 of a number of samples in a hypercube. A discrepancy quantifies the 

210 distance between the continuous uniform distribution on a hypercube and the 

211 discrete uniform distribution on :math:`n` distinct sample points. 

212 

213 The lower the value is, the better the coverage of the parameter space is. 

214 

215 For a collection of subsets of the hypercube, the discrepancy is the 

216 difference between the fraction of sample points in one of those 

217 subsets and the volume of that subset. There are different definitions of 

218 discrepancy corresponding to different collections of subsets. Some 

219 versions take a root mean square difference over subsets instead of 

220 a maximum. 

221 

222 A measure of uniformity is reasonable if it satisfies the following 

223 criteria [1]_: 

224 

225 1. It is invariant under permuting factors and/or runs. 

226 2. It is invariant under rotation of the coordinates. 

227 3. It can measure not only uniformity of the sample over the hypercube, 

228 but also the projection uniformity of the sample over non-empty 

229 subset of lower dimension hypercubes. 

230 4. There is some reasonable geometric meaning. 

231 5. It is easy to compute. 

232 6. It satisfies the Koksma-Hlawka-like inequality. 

233 7. It is consistent with other criteria in experimental design. 

234 

235 Four methods are available: 

236 

237 * ``CD``: Centered Discrepancy - subspace involves a corner of the 

238 hypercube 

239 * ``WD``: Wrap-around Discrepancy - subspace can wrap around bounds 

240 * ``MD``: Mixture Discrepancy - mix between CD/WD covering more criteria 

241 * ``L2-star``: L2-star discrepancy - like CD BUT variant to rotation 

242 

243 See [2]_ for precise definitions of each method. 

244 

245 Lastly, using ``iterative=True``, it is possible to compute the 

246 discrepancy as if we had :math:`n+1` samples. This is useful if we want 

247 to add a point to a sampling and check the candidate which would give the 

248 lowest discrepancy. Then you could just update the discrepancy with 

249 each candidate using `update_discrepancy`. This method is faster than 

250 computing the discrepancy for a large number of candidates. 

251 

252 References 

253 ---------- 

254 .. [1] Fang et al. "Design and modeling for computer experiments". 

255 Computer Science and Data Analysis Series, 2006. 

256 .. [2] Zhou Y.-D. et al. Mixture discrepancy for quasi-random point sets. 

257 Journal of Complexity, 29 (3-4) , pp. 283-301, 2013. 

258 .. [3] T. T. Warnock. "Computational investigations of low discrepancy 

259 point sets". Applications of Number Theory to Numerical 

260 Analysis, Academic Press, pp. 319-343, 1972. 

261 

262 Examples 

263 -------- 

264 Calculate the quality of the sample using the discrepancy: 

265 

266 >>> from scipy.stats import qmc 

267 >>> space = np.array([[1, 3], [2, 6], [3, 2], [4, 5], [5, 1], [6, 4]]) 

268 >>> l_bounds = [0.5, 0.5] 

269 >>> u_bounds = [6.5, 6.5] 

270 >>> space = qmc.scale(space, l_bounds, u_bounds, reverse=True) 

271 >>> space 

272 array([[0.08333333, 0.41666667], 

273 [0.25 , 0.91666667], 

274 [0.41666667, 0.25 ], 

275 [0.58333333, 0.75 ], 

276 [0.75 , 0.08333333], 

277 [0.91666667, 0.58333333]]) 

278 >>> qmc.discrepancy(space) 

279 0.008142039609053464 

280 

281 We can also compute iteratively the ``CD`` discrepancy by using 

282 ``iterative=True``. 

283 

284 >>> disc_init = qmc.discrepancy(space[:-1], iterative=True) 

285 >>> disc_init 

286 0.04769081147119336 

287 >>> qmc.update_discrepancy(space[-1], space[:-1], disc_init) 

288 0.008142039609053513 

289 

290 """ 

291 sample = np.asarray(sample, dtype=np.float64, order="C") 

292 

293 # Checking that sample is within the hypercube and 2D 

294 if not sample.ndim == 2: 

295 raise ValueError("Sample is not a 2D array") 

296 

297 if not (np.all(sample >= 0) and np.all(sample <= 1)): 

298 raise ValueError("Sample is not in unit hypercube") 

299 

300 workers = _validate_workers(workers) 

301 

302 methods = { 

303 "CD": _cy_wrapper_centered_discrepancy, 

304 "WD": _cy_wrapper_wrap_around_discrepancy, 

305 "MD": _cy_wrapper_mixture_discrepancy, 

306 "L2-star": _cy_wrapper_l2_star_discrepancy, 

307 } 

308 

309 if method in methods: 

310 return methods[method](sample, iterative, workers=workers) 

311 else: 

312 raise ValueError(f"{method!r} is not a valid method. It must be one of" 

313 f" {set(methods)!r}")

376def _perturb_discrepancy(sample: np.ndarray, i1: int, i2: int, k: int, 

377 disc: float): 

378 """Centered discrepancy after an elementary perturbation of a LHS. 

379 

380 An elementary perturbation consists of an exchange of coordinates between 

381 two points: ``sample[i1, k] <-> sample[i2, k]``. By construction, 

382 this operation conserves the LHS properties. 

383 

384 Parameters 

385 ---------- 

386 sample : array_like (n, d) 

387 The sample (before permutation) to compute the discrepancy from. 

388 i1 : int 

389 The first line of the elementary permutation. 

390 i2 : int 

391 The second line of the elementary permutation. 

392 k : int 

393 The column of the elementary permutation. 

394 disc : float 

395 Centered discrepancy of the design before permutation. 

396 

397 Returns 

398 ------- 

399 discrepancy : float 

400 Centered discrepancy of the design after permutation. 

401 

402 References 

403 ---------- 

404 .. [1] Jin et al. "An efficient algorithm for constructing optimal design 

405 of computer experiments", Journal of Statistical Planning and 

406 Inference, 2005. 

407 

408 """ 

409 n = sample.shape[0] 

410 

411 z_ij = sample - 0.5 

412 

413 # Eq (19) 

414 c_i1j = (1. / n ** 2. 

415 * np.prod(0.5 * (2. + abs(z_ij[i1, :]) 

416 + abs(z_ij) - abs(z_ij[i1, :] - z_ij)), axis=1)) 

417 c_i2j = (1. / n ** 2. 

418 * np.prod(0.5 * (2. + abs(z_ij[i2, :]) 

419 + abs(z_ij) - abs(z_ij[i2, :] - z_ij)), axis=1)) 

420 

421 # Eq (20) 

422 c_i1i1 = (1. / n ** 2 * np.prod(1 + abs(z_ij[i1, :])) 

423 - 2. / n * np.prod(1. + 0.5 * abs(z_ij[i1, :]) 

424 - 0.5 * z_ij[i1, :] ** 2)) 

425 c_i2i2 = (1. / n ** 2 * np.prod(1 + abs(z_ij[i2, :])) 

426 - 2. / n * np.prod(1. + 0.5 * abs(z_ij[i2, :]) 

427 - 0.5 * z_ij[i2, :] ** 2)) 

428 

429 # Eq (22), typo in the article in the denominator i2 -> i1 

430 num = (2 + abs(z_ij[i2, k]) + abs(z_ij[:, k]) 

431 - abs(z_ij[i2, k] - z_ij[:, k])) 

432 denum = (2 + abs(z_ij[i1, k]) + abs(z_ij[:, k]) 

433 - abs(z_ij[i1, k] - z_ij[:, k])) 

434 gamma = num / denum 

435 

436 # Eq (23) 

437 c_p_i1j = gamma * c_i1j 

438 # Eq (24) 

439 c_p_i2j = c_i2j / gamma 

440 

441 alpha = (1 + abs(z_ij[i2, k])) / (1 + abs(z_ij[i1, k])) 

442 beta = (2 - abs(z_ij[i2, k])) / (2 - abs(z_ij[i1, k])) 

443 

444 g_i1 = np.prod(1. + abs(z_ij[i1, :])) 

445 g_i2 = np.prod(1. + abs(z_ij[i2, :])) 

446 h_i1 = np.prod(1. + 0.5 * abs(z_ij[i1, :]) - 0.5 * (z_ij[i1, :] ** 2)) 

447 h_i2 = np.prod(1. + 0.5 * abs(z_ij[i2, :]) - 0.5 * (z_ij[i2, :] ** 2)) 

448 

449 # Eq (25), typo in the article g is missing 

450 c_p_i1i1 = ((g_i1 * alpha) / (n ** 2) - 2. * alpha * beta * h_i1 / n) 

451 # Eq (26), typo in the article n ** 2 

452 c_p_i2i2 = ((g_i2 / ((n ** 2) * alpha)) - (2. * h_i2 / (n * alpha * beta))) 

453 

454 # Eq (26) 

455 sum_ = c_p_i1j - c_i1j + c_p_i2j - c_i2j 

456 

457 mask = np.ones(n, dtype=bool) 

458 mask[[i1, i2]] = False 

459 sum_ = sum(sum_[mask]) 

460 

461 disc_ep = (disc + c_p_i1i1 - c_i1i1 + c_p_i2i2 - c_i2i2 + 2 * sum_) 

462 

463 return disc_ep

537def van_der_corput(

542 scramble: bool = False, 

543 seed: SeedType = None,

545 """Van der Corput sequence. 

546 

547 Pseudo-random number generator based on a b-adic expansion. 

548 

549 Scrambling uses permutations of the remainders (see [1]_). Multiple 

550 permutations are applied to construct a point. The sequence of 

551 permutations has to be the same for all points of the sequence. 

552 

553 Parameters 

554 ---------- 

555 n : int 

556 Number of element of the sequence. 

557 base : int, optional 

558 Base of the sequence. Default is 2. 

559 start_index : int, optional 

560 Index to start the sequence from. Default is 0. 

561 scramble : bool, optional 

562 If True, use Owen scrambling. Otherwise no scrambling is done. 

563 Default is True. 

564 seed : {None, int, `numpy.random.Generator`}, optional 

565 If `seed` is None the `numpy.random.Generator` singleton is used. 

566 If `seed` is an int, a new ``Generator`` instance is used, 

567 seeded with `seed`. 

568 If `seed` is already a ``Generator`` instance then that instance is 

569 used. 

570 workers : int, optional 

571 Number of workers to use for parallel processing. If -1 is 

572 given all CPU threads are used. Default is 1. 

573 

574 Returns 

575 ------- 

576 sequence : list (n,) 

577 Sequence of Van der Corput. 

578 

579 References 

580 ---------- 

581 .. [1] A. B. Owen. "A randomized Halton algorithm in R", 

582 arXiv:1706.02808, 2017. 

583 

584 """ 

585 if base < 2: 

586 raise ValueError("'base' must be at least 2") 

587 

588 if scramble: 

589 rng = check_random_state(seed) 

590 # In Algorithm 1 of Owen 2017, a permutation of `np.arange(base)` is 

591 # created for each positive integer `k` such that `1 - base**-k < 1` 

592 # using floating-point arithmetic. For double precision floats, the 

593 # condition `1 - base**-k < 1` can also be written as `base**-k > 

594 # 2**-54`, which makes it more apparent how many permutations we need 

595 # to create. 

596 count = math.ceil(54 / math.log2(base)) - 1

603

885class LatinHypercube(QMCEngine): 

886 r"""Latin hypercube sampling (LHS). 

887 

888 A Latin hypercube sample [1]_ generates :math:`n` points in 

889 :math:`[0,1)^{d}`. Each univariate marginal distribution is stratified, 

890 placing exactly one point in :math:`[j/n, (j+1)/n)` for 

891 :math:`j=0,1,...,n-1`. They are still applicable when :math:`n << d`. 

892 

893 Parameters 

894 ---------- 

895 d : int 

896 Dimension of the parameter space. 

897 centered : bool, optional 

898 Center the point within the multi-dimensional grid. Default is False. 

899 optimization : {None, "random-cd"}, optional 

900 Whether to use an optimization scheme to construct a LHS. 

901 Default is None. 

902 

903 * ``random-cd``: random permutations of coordinates to lower the 

904 centered discrepancy [5]_. The best design based on the centered 

905 discrepancy is constantly updated. Centered discrepancy-based 

906 design shows better space filling robustness toward 2D and 3D 

907 subprojections compared to using other discrepancy measures [6]_. 

908 

909 .. versionadded:: 1.8.0 

910 

911 strength : {1, 2}, optional 

912 Strength of the LHS. ``strength=1`` produces a plain LHS while 

913 ``strength=2`` produces an orthogonal array based LHS of strength 2 

914 [7]_, [8]_. In that case, only ``n=p**2`` points can be sampled, 

915 with ``p`` a prime number. It also constrains ``d <= p + 1``. 

916 Default is 1. 

917 

918 .. versionadded:: 1.8.0 

919 

920 seed : {None, int, `numpy.random.Generator`}, optional 

921 If `seed` is None the `numpy.random.Generator` singleton is used. 

922 If `seed` is an int, a new ``Generator`` instance is used, 

923 seeded with `seed`. 

924 If `seed` is already a ``Generator`` instance then that instance is 

925 used. 

926 

927 Notes 

928 ----- 

929 

930 When LHS is used for integrating a function :math:`f` over :math:`n`, 

931 LHS is extremely effective on integrands that are nearly additive [2]_. 

932 With a LHS of :math:`n` points, the variance of the integral is always 

933 lower than plain MC on :math:`n-1` points [3]_. There is a central limit 

934 theorem for LHS on the mean and variance of the integral [4]_, but not 

935 necessarily for optimized LHS due to the randomization. 

936 

937 :math:`A` is called an orthogonal array of strength :math:`t` if in each 

938 n-row-by-t-column submatrix of :math:`A`: all :math:`p^t` possible 

939 distinct rows occur the same number of times. The elements of :math:`A` 

940 are in the set :math:`\{0, 1, ..., p-1\}`, also called symbols. 

941 The constraint that :math:`p` must be a prime number is to allow modular 

942 arithmetic. 

943 

944 Strength 1 (plain LHS) brings an advantage over strength 0 (MC) and 

945 strength 2 is a useful increment over strength 1. Going to strength 3 is 

946 a smaller increment and scrambled QMC like Sobol', Halton are more 

947 performant [7]_. 

948 

949 To create a LHS of strength 2, the orthogonal array :math:`A` is 

950 randomized by applying a random, bijective map of the set of symbols onto 

951 itself. For example, in column 0, all 0s might become 2; in column 1, 

952 all 0s might become 1, etc. 

953 Then, for each column :math:`i` and symbol :math:`j`, we add a plain, 

954 one-dimensional LHS of size :math:`p` to the subarray where 

955 :math:`A^i = j`. The resulting matrix is finally divided by :math:`p`.

1237 There are many versions of Sobol' sequences depending on their 

1238 'direction numbers'. This code uses direction numbers from [3]_. Hence, 

1239 the maximum number of dimension is 21201. The direction numbers have been 

1240 precomputed with search criterion 6 and can be retrieved at 

1241 https://web.maths.unsw.edu.au/~fkuo/sobol/. 

1242 

1243 .. warning:: 

1244 

1245 Sobol' sequences are a quadrature rule and they lose their balance 

1246 properties if one uses a sample size that is not a power of 2, or skips 

1247 the first point, or thins the sequence [4]_. 

1248 

1249 If :math:`n=2^m` points are not enough then one should take :math:`2^M` 

1250 points for :math:`M>m`. When scrambling, the number R of independent 

1251 replicates does not have to be a power of 2. 

1252 

1253 Sobol' sequences are generated to some number :math:`B` of bits. 

1254 After :math:`2^B` points have been generated, the sequence will repeat. 

1255 Currently :math:`B=30`.

1503 inv_transform: bool = True, 

1504 engine: Optional[QMCEngine] = None, 

1505 seed: SeedType = None 

1506 ) -> None:

1508 d = mean.shape[0] 

1509 if cov is not None: 

1510 # covariance matrix provided 

1511 cov = np.array(cov, copy=False, ndmin=2) 

1512 # check for square/symmetric cov matrix and mean vector has the 

1513 # same d 

1514 if not mean.shape[0] == cov.shape[0]: 

1515 raise ValueError("Dimension mismatch between mean and " 

1516 "covariance.") 

1517 if not np.allclose(cov, cov.transpose()): 

1518 raise ValueError("Covariance matrix is not symmetric.") 

1519 # compute Cholesky decomp; if it fails, do the eigen decomposition 

1520 try: 

1521 cov_root = np.linalg.cholesky(cov).transpose() 

1522 except np.linalg.LinAlgError:

1524 if not np.all(eigval >= -1.0e-8): 

1525 raise ValueError("Covariance matrix not PSD.") 

1526 eigval = np.clip(eigval, 0.0, None) 

1527 cov_root = (eigvec * np.sqrt(eigval)).transpose() 

1528 elif cov_root is not None: 1528 ↛ 1530line 1528 didn't jump to line 1530, because the condition on line 1528 was never true

1529 # root decomposition provided 

1530 cov_root = np.atleast_2d(cov_root) 

1531 if not mean.shape[0] == cov_root.shape[0]: 

1532 raise ValueError("Dimension mismatch between mean and " 

1533 "covariance.") 

1534 else: 

1535 # corresponds to identity covariance matrix 

1536 cov_root = None

1539 self._inv_transform = inv_transform 

1540

1544 else:

1555 "`scipy.stats.qmc.QMCEngine` or `None`.")

1591 def _correlate(self, base_samples: np.ndarray) -> np.ndarray: 

1592 if self._corr_matrix is not None:

1598 def _standard_normal_samples(self, n: IntNumber = 1) -> np.ndarray: 

1599 """Draw `n` QMC samples from the standard Normal :math:`N(0, I_d)`. 

1600 

1601 Parameters 

1602 ---------- 

1603 n : int, optional 

1604 Number of samples to generate in the parameter space. Default is 1. 

1605 

1606 Returns 

1607 ------- 

1608 sample : array_like (n, d) 

1609 Sample. 

1610 

1611 """ 

1612 # get base samples 

1613 samples = self.engine.random(n) 

1614 if self._inv_transform: 

1615 # apply inverse transform 

1616 # (values to close to 0/1 result in inf values) 

1617 return stats.norm.ppf(0.5 + (1 - 1e-10) * (samples - 0.5)) # type: ignore[attr-defined] 

1618 else: 

1619 # apply Box-Muller transform (note: indexes starting from 1) 

1620 even = np.arange(0, samples.shape[-1], 2) 

1621 Rs = np.sqrt(-2 * np.log(samples[:, even])) 

1622 thetas = 2 * math.pi * samples[:, 1 + even] 

1623 cos = np.cos(thetas) 

1624 sin = np.sin(thetas) 

1625 transf_samples = np.stack([Rs * cos, Rs * sin], 

1626 -1).reshape(n, -1) 

1627 # make sure we only return the number of dimension requested 

1628 return transf_samples[:, : self.d] # type: ignore[misc]
