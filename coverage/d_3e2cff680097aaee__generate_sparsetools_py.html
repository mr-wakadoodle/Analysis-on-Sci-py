197def get_thunk_type_set(): 

198 """ 

199 Get a list containing cartesian product of data types, plus a getter routine. 

200 

201 Returns 

202 ------- 

203 i_types : list [(j, I_typenum, None, I_type, None), ...] 

204 Pairing of index type numbers and the corresponding C++ types, 

205 and an unique index `j`. This is for routines that are parameterized 

206 only by I but not by T. 

207 it_types : list [(j, I_typenum, T_typenum, I_type, T_type), ...] 

208 Same as `i_types`, but for routines parameterized both by T and I. 

209 getter_code : str 

210 C++ code for a function that takes I_typenum, T_typenum and returns 

211 the unique index corresponding to the lists, or -1 if no match was 

212 found. 

213 

214 """ 

215 it_types = [] 

216 i_types = [] 

217 

218 j = 0 

219 

220 getter_code = " if (0) {}" 

221 

222 for I_typenum, I_type in I_TYPES: 

223 piece = """ 

224 else if (I_typenum == %(I_typenum)s) { 

225 if (T_typenum == -1) { return %(j)s; }""" 

226 getter_code += piece % dict(I_typenum=I_typenum, j=j) 

227 

228 i_types.append((j, I_typenum, None, I_type, None)) 

229 j += 1 

230 

231 for T_typenum, T_type in T_TYPES: 

232 piece = """ 

233 else if (T_typenum == %(T_typenum)s) { return %(j)s; }""" 

234 getter_code += piece % dict(T_typenum=T_typenum, j=j) 

235 

236 it_types.append((j, I_typenum, T_typenum, I_type, T_type)) 

237 j += 1 

238 

239 getter_code += """ 

240 }""" 

241 

242 return i_types, it_types, GET_THUNK_CASE_TEMPLATE % dict(content=getter_code)

245def parse_routine(name, args, types): 

246 """ 

247 Generate thunk and method code for a given routine. 

248 

249 Parameters 

250 ---------- 

251 name : str 

252 Name of the C++ routine 

253 args : str 

254 Argument list specification (in format explained above) 

255 types : list 

256 List of types to instantiate, as returned `get_thunk_type_set` 

257 

258 """ 

259 

260 ret_spec = args[0] 

261 arg_spec = args[1:] 

262 

263 def get_arglist(I_type, T_type): 

264 """ 

265 Generate argument list for calling the C++ function 

266 """ 

267 args = [] 

268 next_is_writeable = False 

269 j = 0 

270 for t in arg_spec: 

271 const = '' if next_is_writeable else 'const ' 

272 next_is_writeable = False 

273 if t == '*': 

274 next_is_writeable = True 

275 continue 

276 elif t == 'i': 

277 args.append("*(%s*)a[%d]" % (const + I_type, j)) 

278 elif t == 'I': 

279 args.append("(%s*)a[%d]" % (const + I_type, j)) 

280 elif t == 'T': 

281 args.append("(%s*)a[%d]" % (const + T_type, j)) 

282 elif t == 'B': 

283 args.append("(npy_bool_wrapper*)a[%d]" % (j,)) 

284 elif t == 'V': 

285 if const: 

286 raise ValueError("'V' argument must be an output arg") 

287 args.append("(std::vector<%s>*)a[%d]" % (I_type, j,)) 

288 elif t == 'W': 

289 if const: 

290 raise ValueError("'W' argument must be an output arg") 

291 args.append("(std::vector<%s>*)a[%d]" % (T_type, j,)) 

292 elif t == 'l': 

293 args.append("*(%snpy_int64*)a[%d]" % (const, j)) 

294 else: 

295 raise ValueError("Invalid spec character %r" % (t,)) 

296 j += 1 

297 return ", ".join(args) 

298 

299 # Generate thunk code: a giant switch statement with different 

300 # type combinations inside. 

301 thunk_content = """int j = get_thunk_case(I_typenum, T_typenum); 

302 switch (j) {""" 

303 for j, I_typenum, T_typenum, I_type, T_type in types: 

304 arglist = get_arglist(I_type, T_type) 

305 

306 piece = """ 

307 case %(j)s:""" 

308 if ret_spec == 'v': 

309 piece += """ 

310 (void)%(name)s(%(arglist)s); 

311 return 0;""" 

312 else: 

313 piece += """ 

314 return %(name)s(%(arglist)s);""" 

315 thunk_content += piece % dict(j=j, I_type=I_type, T_type=T_type, 

316 I_typenum=I_typenum, T_typenum=T_typenum, 

317 arglist=arglist, name=name) 

318 

319 thunk_content += """ 

320 default: 

321 throw std::runtime_error("internal error: invalid argument typenums"); 

322 }""" 

323 

324 thunk_code = THUNK_TEMPLATE % dict(name=name, 

325 thunk_content=thunk_content) 

326 

327 # Generate method code 

328 method_code = METHOD_TEMPLATE % dict(name=name, 

329 ret_spec=ret_spec, 

330 arg_spec=arg_spec) 

331 

332 return thunk_code, method_code

335def main(): 

336 p = optparse.OptionParser(usage=(__doc__ or '').strip()) 

337 p.add_option("--no-force", action="store_false", 

338 dest="force", default=True) 

339 options, args = p.parse_args() 

340 

341 names = [] 

342 

343 i_types, it_types, getter_code = get_thunk_type_set() 

344 

345 # Generate *_impl.h for each compilation unit 

346 for unit_name, routines in COMPILATION_UNITS: 

347 thunks = [] 

348 methods = [] 

349 

350 # Generate thunks and methods for all routines 

351 for line in routines.splitlines(): 

352 line = line.strip() 

353 if not line or line.startswith('#'): 

354 continue 

355 

356 try: 

357 name, args = line.split(None, 1) 

358 except ValueError as e: 

359 raise ValueError("Malformed line: %r" % (line,)) from e 

360 

361 args = "".join(args.split()) 

362 if 't' in args or 'T' in args: 

363 thunk, method = parse_routine(name, args, it_types) 

364 else: 

365 thunk, method = parse_routine(name, args, i_types) 

366 

367 if name in names: 

368 raise ValueError("Duplicate routine %r" % (name,)) 

369 

370 names.append(name) 

371 thunks.append(thunk) 

372 methods.append(method) 

373 

374 # Produce output 

375 dst = os.path.join(os.path.dirname(__file__), 

376 'sparsetools', 

377 unit_name + '_impl.h') 

378 if newer(__file__, dst) or options.force: 

379 print("[generate_sparsetools] generating %r" % (dst,)) 

380 with open(dst, 'w') as f: 

381 write_autogen_blurb(f) 

382 f.write(getter_code) 

383 for thunk in thunks: 

384 f.write(thunk) 

385 for method in methods: 

386 f.write(method) 

387 else: 

388 print("[generate_sparsetools] %r already up-to-date" % (dst,)) 

389 

390 # Generate code for method struct 

391 method_defs = "" 

392 for name in names: 

393 method_defs += "NPY_VISIBILITY_HIDDEN PyObject *%s_method(PyObject *, PyObject *);\n" % (name,) 

394 

395 method_struct = """\nstatic struct PyMethodDef sparsetools_methods[] = {""" 

396 for name in names: 

397 method_struct += """ 

398 {"%(name)s", (PyCFunction)%(name)s_method, METH_VARARGS, NULL},""" % dict(name=name) 

399 method_struct += """ 

400 {NULL, NULL, 0, NULL} 

401 };""" 

402 

403 # Produce sparsetools_impl.h 

404 dst = os.path.join(os.path.dirname(__file__), 

405 'sparsetools', 

406 'sparsetools_impl.h') 

407 

408 if newer(__file__, dst) or options.force: 

409 print("[generate_sparsetools] generating %r" % (dst,)) 

410 with open(dst, 'w') as f: 

411 write_autogen_blurb(f) 

412 f.write(method_defs) 

413 f.write(method_struct) 

414 else: 

415 print("[generate_sparsetools] %r already up-to-date" % (dst,))
