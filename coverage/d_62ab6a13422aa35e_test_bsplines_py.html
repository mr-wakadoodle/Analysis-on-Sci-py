228def test_sepfir2d_invalid_filter(): 

229 filt = np.array([1.0, 2.0, 4.0, 2.0, 1.0]) 

230 image = np.random.rand(7, 9) 

231 # No error for odd lengths 

232 signal.sepfir2d(image, filt, filt[2:]) 

233 

234 # Row or column filter must be odd 

235 with pytest.raises(ValueError, match="odd length"): 

236 signal.sepfir2d(image, filt, filt[1:]) 

237 with pytest.raises(ValueError, match="odd length"): 

238 signal.sepfir2d(image, filt[1:], filt) 

239 

240 # Filters must be 1-dimensional 

241 with pytest.raises(ValueError, match="object too deep"): 

242 signal.sepfir2d(image, filt.reshape(1, -1), filt) 

243 with pytest.raises(ValueError, match="object too deep"): 

244 signal.sepfir2d(image, filt, filt.reshape(1, -1))
