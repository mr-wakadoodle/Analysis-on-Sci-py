65def cases_test_cont_fit(): 

66 # this tests the closeness of the estimated parameters to the true 

67 # parameters with fit method of continuous distributions 

68 # Note: is slow, some distributions don't converge with sample

75@pytest.mark.slow 

76@pytest.mark.parametrize('distname,arg', cases_test_cont_fit()) 

77@pytest.mark.parametrize('method', ["MLE", 'MM']) 

78def test_cont_fit(distname, arg, method): 

79 if distname in failing_fits[method]: 

80 # Skip failing fits unless overridden 

81 try: 

82 xfail = not int(os.environ['SCIPY_XFAIL']) 

83 except Exception: 

84 xfail = True 

85 if xfail: 

86 msg = "Fitting %s doesn't work reliably yet" % distname 

87 msg += (" [Set environment variable SCIPY_XFAIL=1 to run this" 

88 " test nevertheless.]") 

89 pytest.xfail(msg) 

90 

91 distfn = getattr(stats, distname) 

92 

93 truearg = np.hstack([arg, [0.0, 1.0]]) 

94 diffthreshold = np.max(np.vstack([truearg*thresh_percent, 

95 np.full(distfn.numargs+2, thresh_min)]), 

96 0) 

97 

98 for fit_size in fit_sizes: 

99 # Note that if a fit succeeds, the other fit_sizes are skipped 

100 np.random.seed(1234) 

101 

102 with np.errstate(all='ignore'): 

103 rvs = distfn.rvs(size=fit_size, *arg) 

104 est = distfn.fit(rvs, method=method) # start with default values 

105 

106 diff = est - truearg 

107 

108 # threshold for location 

109 diffthreshold[-2] = np.max([np.abs(rvs.mean())*thresh_percent, 

110 thresh_min]) 

111 

112 if np.any(np.isnan(est)): 

113 raise AssertionError('nan returned in fit') 

114 else: 

115 if np.all(np.abs(diff) <= diffthreshold): 

116 break 

117 else: 

118 txt = 'parameter: %s\n' % str(truearg) 

119 txt += 'estimated: %s\n' % str(est) 

120 txt += 'diff : %s\n' % str(diff) 

121 raise AssertionError('fit not very good in %s\n' % distfn.name + txt)
